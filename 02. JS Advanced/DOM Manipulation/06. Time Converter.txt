<script>
    function attachEventsListeners() {
        let days = document.getElementById('daysBtn');
        let hours = document.getElementById('hoursBtn');
        let mins = document.getElementById('minutesBtn');
        let sec = document.getElementById('secondsBtn');

        let dayValue = document.getElementById('days');
        let hourValue = document.getElementById('hours');
        let minValue = document.getElementById('minutes');
        let secValue = document.getElementById('seconds');

        days.addEventListener('click', function () {
            hourValue.value = dayValue.value * 24;
            minValue.value = hourValue.value * 60;
            secValue.value = minValue.value * 60;

        });
        hours.addEventListener('click', function () {
            dayValue.value = hourValue.value / 24;
            minValue.value = hourValue.value * 60;
            secValue.value = minValue.value * 60;
        });
        mins.addEventListener('click', function () {
            dayValue.value = (minValue.value / 60) / 24;
            hourValue.value = minValue.value / 60;
            secValue.value = minValue.value * 60;
        });
        sec.addEventListener('click', function () {
            minValue.value = secValue.value / 60;
            hourValue.value = (secValue.value / 60) / 60;
            dayValue.value = ((secValue.value / 60) / 60) / 24;
        });
    }
</script>
